// ── Utility: shuffle array in place ───────────────────────────────────────

function shuffle(arr) {

  for (let i = arr.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [arr[i], arr[j]] = [arr[j], arr[i]];

  }

  return arr;

}


// ── Session Selection Logic ────────────────────────────────────────────────

// Rules:

//   ≤ 3 attended  → evaluate all of them

//   > 3 attended  → pick exactly 3, guaranteeing ≥1 Day 2 session if any Day 2 was attended

function selectSessions(checkedIndices) {

  const MAX = 3;

  const attended = checkedIndices.map(i => SESSIONS[i]);


  if (attended.length <= MAX) return [...attended];


  const day1 = shuffle(attended.filter(s => s.day === 1));

  const day2 = shuffle(attended.filter(s => s.day === 2));

  const selected = [];


  // Guarantee at least one Day 2 session when any were attended

  if (day2.length > 0) {

    selected.push(day2.shift());       // 1 guaranteed Day 2

    while (selected.length < MAX && day1.length > 0) selected.push(day1.shift());

    while (selected.length < MAX && day2.length > 0) selected.push(day2.shift());

  } else {

    while (selected.length < MAX && day1.length > 0) selected.push(day1.shift());

  }


  // Preserve chronological order for display

  selected.sort((a, b) => a.index - b.index);

  return selected;

}

}

  }

function submitForm() {

  const btnNext = document.getElementById('btnNext');

  btnNext.disabled = true;

  btnNext.innerHTML = '<span class="spinner"></span>Submitting…';


  const payload = flattenData(collectFormData());


  fetch('https://formspree.io/f/' + FORMSPREE_ID, {

    method:  'POST',

    headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },

    body:    JSON.stringify(payload)

  })

  .then(function(response) { return response.json(); })

  .then(function(result) {

    if (result.ok) {

      submitted = true;

      showStep(currentPos + 1);

    } else {

      throw new Error(result.error || 'Submission rejected');

    }

  })

  .catch(function(err) {

    btnNext.disabled = false;

    btnNext.textContent = 'Submit ✓';

    alert('Sorry, there was an error saving your response. Please try again.\n\n' + err.message);

  });

}

    });

  }

}
